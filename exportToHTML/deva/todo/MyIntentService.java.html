package deva.todo; 
 
import android.annotation.TargetApi; 
import android.app.Notification; 
import android.app.NotificationManager; 
import android.app.PendingIntent; 
import android.app.Service; 
import android.app.TaskStackBuilder; 
import android.content.Intent; 
import android.os.Build; 
import android.os.IBinder; 
import android.support.annotation.Nullable; 
import android.util.Log; 
 
import java.io.BufferedReader; 
import java.io.File; 
import java.io.FileInputStream; 
 
import java.io.FileOutputStream; 
import java.io.IOException; 
import java.io.InputStreamReader; 
import java.text.DateFormat; 
import java.text.SimpleDateFormat; 
import java.util.Calendar; 
import java.util.concurrent.TimeUnit; 
 
public class MyIntentService extends Service  { 
    public String lol; 
 
    final class TheThread implements Runnable { 
        int service_number; 
        TheThread (int serviceId) 
        { 
            this.service_number = serviceId; 
        } 
 
        @TargetApi(Build.VERSION_CODES.JELLY_BEAN) 
        @Override 
        public void run() { 
 
            DateFormat df = new SimpleDateFormat("EEE, d MMM yyyy HH:mm:ss Z");     //Wed, 4 Jul 2008 12:08:56 -0530 
            String date = df.format(Calendar.getInstance().getTime()); 
 
            int i,spinner_number; 
            String remind; 
            File directory,myFile; 
            FileOutputStream fos; 
            try { 
 
                directory = new File("/sdcard/logs"); 
                if  (!directory.exists()){ 
                    directory.mkdir(); 
                } 
                myFile = new File("/sdcard/logs/todo_log.txt"); 
                myFile.createNewFile(); 
                fos =  new   FileOutputStream (myFile,true); 
                fos.write(("\n\n"+date).getBytes()); 
                fos.write("\nProcess Started.".getBytes()); 
 
                FileInputStream f13 = openFileInput("row1.txt"); 
                InputStreamReader ias3 = new InputStreamReader(f13); 
                BufferedReader br3 = new BufferedReader(ias3); 
                lol = br3.readLine(); 
                f13.close(); 
 
                FileInputStream fileInputStream = openFileInput("spinner.txt"); 
                InputStreamReader inputStreamReader = new InputStreamReader(fileInputStream); 
                BufferedReader bufferedReader = new BufferedReader(inputStreamReader); 
                spinner_number = Integer.parseInt(bufferedReader.readLine()) ; 
                fileInputStream.close(); 
                fos.write(("\nReminder value : "+spinner_number).getBytes()); 
 
                fos.write("\nWaiting...".getBytes()); 
                if(spinner_number<=30) { 
                    synchronized (this) { 
                        TimeUnit.MINUTES.sleep(spinner_number); 
                    } 
                } 
                else if (spinner_number>30 && spinner_number<=60) 
                { 
                    synchronized (this) { 
                        TimeUnit.MINUTES.sleep(spinner_number-30); 
                        TimeUnit.MINUTES.sleep(30); 
                    } 
                } 
                else 
                { 
                    synchronized (this) { 
                        TimeUnit.MINUTES.sleep(30); 
                    } 
                } 
                String date2 = df.format(Calendar.getInstance().getTime()); 
                fos.write(("\n"+date2).getBytes()); 
                fos.write("\nWait completed.".getBytes()); 
 
                FileInputStream f1 = openFileInput("data1.txt"); 
                InputStreamReader ias = new InputStreamReader(f1); 
                BufferedReader br = new BufferedReader(ias); 
                StringBuilder stringBuffer = new StringBuilder(); 
                i=1; 
                while (i<8) { 
                    stringBuffer.append(br.readLine()).append(" "); 
                    i++; 
                } 
                remind = stringBuffer.toString(); 
 
                Intent intent = new Intent(MyIntentService.this, MainActivity.class); 
                TaskStackBuilder stackBuilder = TaskStackBuilder.create(MyIntentService.this); 
                stackBuilder.addParentStack(MainActivity.class); 
                stackBuilder.addNextIntent(intent); 
                PendingIntent pIn = stackBuilder.getPendingIntent(0,PendingIntent.FLAG_UPDATE_CURRENT); 
                Notification notification = new Notification.Builder(MyIntentService.this) 
                        .setTicker("Hey there!") 
                        .setContentTitle("Reminder!") 
                        .setAutoCancel(true) 
                        .setPriority(Notification.PRIORITY_HIGH) 
                        .setDefaults(Notification.DEFAULT_LIGHTS|Notification.DEFAULT_VIBRATE|Notification.DEFAULT_SOUND) 
                        .setContentText(""+remind) 
                        .setSmallIcon(R.mipmap.ic_launcher) 
                        .setContentIntent(pIn) 
                        .build(); 
                NotificationManager nm = (NotificationManager) getSystemService(NOTIFICATION_SERVICE); 
 
                FileInputStream fu = openFileInput("row1.txt"); 
                InputStreamReader iu = new InputStreamReader(fu); 
                BufferedReader bu = new BufferedReader(iu); 
                lol = bu.readLine(); 
                fu.close(); 
 
                if (lol.equals("1")) { 
                    nm.notify(0, notification); 
                    fos.write("\nNotified!".getBytes()); 
                } 
                else { 
                    fos.write("\nNOT Notified.".getBytes()); 
                } 
 
                fos.write("\nProcess completed.".getBytes()); 
                fos.close(); 
 
            } catch (InterruptedException | IOException e) { 
                e.printStackTrace(); 
                Log.d("debug","At catch() of TheThread."); 
                try { 
                    FileOutputStream fileOutputStream = openFileOutput("spinner.txt",MODE_PRIVATE); 
                    fileOutputStream.write(("25").getBytes()); 
                    fileOutputStream.close(); 
                } catch (IOException e1) { 
                    e1.printStackTrace(); 
                } 
            } 
            stopSelf(this.service_number); 
        } 
    } 
 
 
    @Override 
    public int onStartCommand(Intent intent, int flags, int startId) { 
        Thread thr = new Thread(new TheThread(startId)); 
        thr.start(); 
        return START_STICKY; 
    } 
 
    @Override 
    public void onDestroy() { 
        super.onDestroy(); 
        if(lol.equals("1")) 
        { 
            Intent ino = new Intent(); 
            ino.addFlags(Intent.FLAG_INCLUDE_STOPPED_PACKAGES); 
            ino.setAction("You"); 
            sendBroadcast(ino); 
            Log.d("debug","Broadcast sent using TheService.onDestroy()"); 
        } 
        else 
        { 
            Log.d("debug","Broadcast NOT sent using TheService.onDestroy()"); 
        } 
    } 
 
    @Nullable 
    @Override 
    public IBinder onBind(Intent intent) { 
        return null; 
    } 
}

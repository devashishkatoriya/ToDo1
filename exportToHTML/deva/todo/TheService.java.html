package deva.todo; 
 
import android.annotation.TargetApi; 
import android.app.Notification; 
import android.app.NotificationManager; 
import android.app.PendingIntent; 
import android.app.Service; 
import android.content.Intent; 
import android.os.Build; 
import android.os.IBinder; 
import android.util.Log; 
 
import java.io.BufferedReader; 
import java.io.FileInputStream; 
 
import java.io.FileOutputStream; 
import java.io.IOException; 
import java.io.InputStreamReader; 
 
public class TheService extends Service  { 
    public String lol; 
    public String thrrrr; 
    final class TheThread implements Runnable { 
        int servv; 
        TheThread (int serviceId) 
        { 
            this.servv = serviceId; 
        } 
 
        @TargetApi(Build.VERSION_CODES.JELLY_BEAN) 
        @Override 
        public void run() { 
            Log.d("debug","Inside TheThread.run()"); 
            try { 
 
                FileOutputStream f1 = openFileOutput("thrrr.db", MODE_PRIVATE); 
                f1.write(("1").getBytes()); 
                f1.close(); 
 
                Log.d("debug", "Creating notification..."); 
                Intent inn = new Intent(getApplicationContext(), MainActivity.class); 
                PendingIntent pIn = PendingIntent.getActivity(getApplicationContext(), 0, inn, 0); 
                Notification noti = new Notification.Builder(getApplicationContext()) 
                        .setTicker("Hey there!") 
                        .setContentTitle("Reminder!") 
                        .setContentText("You have tasks remaining to do.") 
                        .setSmallIcon(R.mipmap.ic_launcher) 
                        .setContentIntent(pIn) 
                        .build(); 
                noti.flags |= Notification.FLAG_AUTO_CANCEL; 
                noti.flags |= Notification.PRIORITY_HIGH; 
                noti.flags |= Notification.DEFAULT_LIGHTS; 
                noti.flags |= Notification.DEFAULT_VIBRATE; 
                noti.flags |= Notification.DEFAULT_SOUND; 
                NotificationManager nm = (NotificationManager) getSystemService(NOTIFICATION_SERVICE); 
                Log.d("debug", "...notification successfully created!"); 
 
                Log.d("debug", "synchronize() running..."); 
                synchronized (this) { 
                    wait(15000); 
                } 
                Log.d("debug", "synchronize() completed."); 
 
                Log.d("debug", "Creating FileInputStream..."); 
                FileInputStream f13 = openFileInput("row1.db"); 
                InputStreamReader ias3 = new InputStreamReader(f13); 
                BufferedReader br3 = new BufferedReader(ias3); 
                lol = br3.readLine(); 
                f13.close(); 
                Log.d("debug", "...FileInputStream successfully created and read."); 
 
                if (lol.equals("1")) { 
                    nm.notify(0, noti); 
                    Log.d("debug", "Notification sent."); 
 
                } 
                else 
                { 
                    Log.d("debug", "Notification NOT sent."); 
                } 
                FileOutputStream fi = openFileOutput("thrrr.db", MODE_PRIVATE); 
                fi.write(("0").getBytes()); 
                fi.close(); 
            } catch (InterruptedException | IOException e) { 
                e.printStackTrace(); 
            } 
 
            Log.d("debug","run() Completed."); 
            stopSelf(this.servv); 
            Log.d("debug","TheThread completed."); 
        } 
    } 
 
    @Override 
    public void onCreate() { 
        super.onCreate(); 
 
    } 
 
    @Override 
    public int onStartCommand(Intent intent, int flags, int startId) { 
        Log.d("debug","Inside TheService.onStartCommand()"); 
 
        try { 
            FileInputStream f13 = openFileInput("thrrr.db"); 
            InputStreamReader ias3 = new InputStreamReader(f13); 
            BufferedReader br3 = new BufferedReader(ias3); 
            thrrrr = br3.readLine(); 
        } catch (IOException e) { 
            e.printStackTrace(); 
            thrrrr = "0"; 
        } 
        if (thrrrr.equals("0")) 
        { 
            Thread thr = new Thread(new TheThread(startId)); 
            thr.start(); 
            Log.d("debug", "TheThread started using TheService.onStartCommand()"); 
        } 
        Log.d("debug","TheService.onStartCommand() completed."); 
        return START_STICKY; 
    } 
 
    @Override 
    public void onDestroy() { 
        super.onDestroy(); 
        Log.d("debug","Inside TheService.onDestroy()"); 
        if(lol.equals("1")) 
        { 
            Intent ino = new Intent(); 
            ino.addFlags(Intent.FLAG_INCLUDE_STOPPED_PACKAGES); 
            ino.setAction("You"); 
            sendBroadcast(ino); 
            Log.d("debug","Broadcast sent using TheService.onDestroy()"); 
        } 
        Log.d("debug","TheService.onDestroy() completed"); 
    } 
    @Override 
    public IBinder onBind(Intent intent) { 
        return null; 
    } 
}
